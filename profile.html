    public class Experience {

        private String front;
        private String back;
                            
        Experience(String front, String back){
            this.front = front;
            this.back = back;
        }
                        
        @Override
        public String toString() {
            return front + ":\n" + 
                    " 웹 페이지 잘 만든다. 다 잘함."
                    + "\n" + back + ":\n" +
                    " 자바 잘함. 오라클 더 잘함."; 
        }
    }

    public class Edu {

        private String university;
        private String major;
        private int graduation;
                            
        Edu(String university, String major, 
                int graduation){
            this.university = university;
            this.major = major;
            this.graduation = graduation;
        }
                        
        @Override
        public String toString() {
            return university + " " + major 
                    + "를 " + graduation
                    + "년도에 졸업하였습니다."; 
        }
    }
